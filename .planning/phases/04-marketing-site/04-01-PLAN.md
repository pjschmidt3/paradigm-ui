<task type="auto">
  <name>Task 2: Install and configure Paradigm UI dependencies</name>
  <files>site/package.json, site/src/lib/utils.ts, site/tailwind.config.ts</files>
  <action>
    1. Install required dependencies in site/:
       ```
       npm install clsx tailwind-merge class-variance-authority motion
       ```

    2. Create site/src/lib/utils.ts with cn() utility:
       ```typescript
       import { clsx, type ClassValue } from 'clsx'
       import { twMerge } from 'tailwind-merge'

       export function cn(...inputs: ClassValue[]) {
         return twMerge(clsx(inputs))
       }
       ```

    3. Update site/tsconfig.json paths to include:
       ```json
       "@/src/*": ["../src/*"],
       "@paradigm/*": ["../registry/new-york/ui/*"]
       ```

    4. Update site/tailwind.config.ts content array to include parent components:
       ```typescript
       content: [
         './src/**/*.{js,ts,jsx,tsx,mdx}',
         '../registry/new-york/ui/**/*.{js,ts,jsx,tsx}'
       ]
       ```

    5. Create site/src/components/paradigm.ts barrel export:
       ```typescript
       // Re-export Paradigm UI components for use in marketing site
       export { Hero } from '@paradigm/hero'
       export { Heading } from '@paradigm/heading'
       export { Paragraph } from '@paradigm/paragraph'
       export { Flex, FlexRow, FlexCol } from '@paradigm/flex'
       export { Box } from '@paradigm/box'
       export { Grid } from '@paradigm/grid'
       export { Appear } from '@paradigm/appear'
       export { SocialLinks } from '@paradigm/social-links'
       export { Button } from '@paradigm/button'
       ```
  </action>
  <verify>
    cd site && npm run build completes without errors
    Import test: Create simple test file importing Hero component and verify no TS errors
  </verify>
  <done>Paradigm UI components importable in site/ via @paradigm/* alias, cn() utility available</done>
</task>
